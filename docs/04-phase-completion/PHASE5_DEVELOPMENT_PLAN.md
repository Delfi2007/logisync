## ğŸ“ Database Schema Additions

### **New Tables for AI Features:**

```sql
-- OCR Processing Logs
CREATE TABLE ocr_documents (
  id SERIAL PRIMARY KEY,
  user_id INT REFERENCES users(id),
  document_type VARCHAR(50), -- 'invoice', 'eway_bill', 'product_label'
  image_url TEXT,
  extracted_text TEXT,
  confidence_score DECIMAL(5,2),
  status VARCHAR(20), -- 'processing', 'completed', 'failed'
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE ocr_fields (
  id SERIAL PRIMARY KEY,
  document_id INT REFERENCES ocr_documents(id),
  field_name VARCHAR(100), -- 'customer_name', 'total_amount', etc.
  extracted_value TEXT,
  confidence DECIMAL(5,2),
  manually_corrected BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMP
);

-- AI Recommendations
CREATE TABLE ai_recommendations (
  id SERIAL PRIMARY KEY,
  user_id INT REFERENCES users(id),
  type VARCHAR(50), -- 'warehouse_selection', 'consolidation', 'delay_prediction'
  title TEXT,
  description TEXT,
  potential_savings DECIMAL(10,2),
  confidence_score DECIMAL(5,2),
  status VARCHAR(20), -- 'pending', 'accepted', 'rejected'
  applied_at TIMESTAMP,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Route Performance Data (for ML training)
CREATE TABLE route_analytics (
  id SERIAL PRIMARY KEY,
  route_id INT REFERENCES routes(id),
  actual_duration INT, -- minutes
  actual_cost DECIMAL(10,2),
  traffic_condition VARCHAR(20),
  weather_condition VARCHAR(20),
  delay_minutes INT,
  completed_at TIMESTAMP
);

-- Demand Forecasting Data
CREATE TABLE demand_history (
  id SERIAL PRIMARY KEY,
  product_id INT REFERENCES products(id),
  date DATE,
  quantity_ordered INT,
  quantity_fulfilled INT,
  stockout_occurred BOOLEAN,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Voice Command Logs
CREATE TABLE voice_commands (
  id SERIAL PRIMARY KEY,
  user_id INT REFERENCES users(id),
  command_text TEXT,
  intent VARCHAR(100), -- 'create_order', 'show_shipments', etc.
  entities JSONB, -- extracted parameters
  success BOOLEAN,
  created_at TIMESTAMP DEFAULT NOW()
);
```
